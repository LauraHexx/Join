/**
 * Changes the z-index property after a delay. So it is possible to click on the elements
 */
function changeZindexAnimation() {
  setTimeout(() => {
    let animationDiv = document.getElementById("animationDiv");
    animationDiv.style.zIndex = 0;
  }, 1500);
}
